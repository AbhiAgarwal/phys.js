suvat.create = function (s, u, v, a, t) {
  if (s != null) {
    equation.s = s;
  }
  if (u != null) {
    equation.u = u;
  }
  if (v != null) {
    equation.v = v;
  }
  if (a != null) {
    equation.a = a;
  }
  if (t != null) {
    equation.t = t;
  }
};
